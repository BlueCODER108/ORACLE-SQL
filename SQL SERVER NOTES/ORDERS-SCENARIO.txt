USE ABCBANK

CREATE TABLE COMPANY
(
CMPID  VARCHAR(10),
CONSTRAINT PK_CMPID_COMPANY PRIMARY KEY(CMPID),
CMPNAME  VARCHAR(20) 
CONSTRAINT NN_CMPNAME_COMPANY NOT NULL
);

CREATE TABLE PRODUCT
(
PID  VARCHAR(10),
CONSTRAINT PK_PID_PRODUCT PRIMARY KEY(PID),
PNAME  VARCHAR(14),
PRICE  MONEY,
CMPID VARCHAR(10),
CONSTRAINT FK_CMPID_PRODUCT FOREIGN KEY(CMPID) 
REFERENCES COMPANY(CMPID)
);
CREATE TABLE CUSTOMER
(
CID  INT,
CONSTRAINT PK_CID_CUSTOMER PRIMARY KEY(CID),
CNAME  VARCHAR(14),
PHNO  BIGINT,
CONSTRAINT UK_PHNO_CUSTOMER UNIQUE(PHNO)
);

CREATE TABLE ORDERS
(
ORDER_ID  INT,
CONSTRAINT PK_ORDID_ORDERS PRIMARY KEY(ORDER_ID),
ORD_DT  DATE CONSTRAINT NN_ORDDT_ORDERS NOT NULL,
QTY INT,
PAYMENT MONEY,
PID  VARCHAR(10),
CONSTRAINT FK_PID_ORDERS FOREIGN KEY(PID)
REFERENCES PRODUCT(PID),
CID  INT,
CONSTRAINT FK_CID_ORDERS FOREIGN KEY(CID)
REFERENCES CUSTOMER(CID)
);
